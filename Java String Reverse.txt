import java.util.*;
public class Main{
    public static void main(String[] args ){
        Scanner sc=new Scanner(System.in);
        String a=sc.nextLine();
        StringBuilder sb=new StringBuilder(a);
        sb=sb.reverse();
        if(a.equals(sb.toString())){
            System.out.print("Yes");
        }
        else{
            System.out.print("No");
        }
    }
}
